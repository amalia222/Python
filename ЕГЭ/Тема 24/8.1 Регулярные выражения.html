    print(match.group(2)) # 10 (–≤—Ç–æ—Ä–∞—è –≥—Ä—É–ø–ø–∞)

2. –ù–µ–∑–∞—Ö–≤–∞—Ç—ã–≤–∞—é—â–∏–µ –≥—Ä—É–ø–ø—ã (Non-capturing Groups)

–ï—Å–ª–∏ —Å–∫–æ–±–∫–∏ –Ω—É–∂–Ω—ã —Ç–æ–ª—å–∫–æ –¥–ª—è –ø—Ä–∏–º–µ–Ω–µ–Ω–∏—è –∫–≤–∞–Ω—Ç–∏—Ñ–∏–∫–∞—Ç–æ—Ä–∞ –∫–æ –≤—Å–µ–π —Ñ—Ä–∞–∑–µ –∏–ª–∏ –≤—ã–±–æ—Ä–∞ –∏–∑ –≤–∞—Ä–∏–∞–Ω—Ç–æ–≤ (—á–µ—Ä–µ–∑ |), –Ω–æ —Å–æ—Ö—Ä–∞–Ω—è—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç –Ω–µ —Ç—Ä–µ–±—É–µ—Ç—Å—è, –∏—Å–ø–æ–ª—å–∑—É–π—Ç–µ —Å–∏–Ω—Ç–∞–∫—Å–∏—Å (?:...).

–ü—Ä–µ–∏–º—É—â–µ—Å—Ç–≤–∞:

  • –≠–∫–æ–Ω–æ–º–∏—Ç –ø–∞–º—è—Ç—å –∏ —Ä–µ—Å—É—Ä—Å—ã, —Ç–∞–∫ –∫–∞–∫ –¥–≤–∏–∂–æ–∫ –Ω–µ –∑–∞–ø–æ–º–∏–Ω–∞–µ—Ç –ø–æ–¥—Å—Ç—Ä–æ–∫—É.
  • –ù–µ —Å–±–∏–≤–∞–µ—Ç –Ω—É–º–µ—Ä–∞—Ü–∏—é –∑–∞—Ö–≤–∞—Ç—ã–≤–∞—é—â–∏—Ö –≥—Ä—É–ø–ø.
# –ò—â–µ–º "batman" –∏–ª–∏ "batmobile", –Ω–æ –Ω–µ —Å–æ—Ö—Ä–∞–Ω—è–µ–º "man" –∏–ª–∏ "mobile" –æ—Ç–¥–µ–ª—å–Ω–æ
pattern = r"bat(?:man|mobile)"
text = "batman"
match = re.match(pattern, text)
if match:
    print(match.group())  # batman
    print(match.groups()) # () - –ø—É—Å—Ç–æ–π –∫–æ—Ä—Ç–µ–∂, —Ç–∞–∫ –∫–∞–∫ –≥—Ä—É–ø–ø–∞ –Ω–µ –∑–∞—Ö–≤–∞—Ç—ã–≤–∞—é—â–∞—è

3. –û–±—Ä–∞—Ç–Ω—ã–µ —Å—Å—ã–ª–∫–∏ (Backreferences)

–ü–æ–∑–≤–æ–ª—è—é—Ç —Å—Å—ã–ª–∞—Ç—å—Å—è –Ω–∞ —Ä–∞–Ω–µ–µ –Ω–∞–π–¥–µ–Ω–Ω—É—é –≥—Ä—É–ø–ø—É –≤–Ω—É—Ç—Ä–∏ —Ç–æ–≥–æ –∂–µ —Ä–µ–≥—É–ª—è—Ä–Ω–æ–≥–æ –≤—ã—Ä–∞–∂–µ–Ω–∏—è: \1, \2 –∏ —Ç.–¥.

# –ü–æ–∏—Å–∫ –ø–æ–≤—Ç–æ—Ä—è—é—â–∏—Ö—Å—è —Å–ª–æ–≤
pattern = r"(\w+) \1"
text = "hello hello world"
match = re.search(pattern, text)
if match:
    print(match.group()) # hello hello

4. –ü—Ä–æ–≤–µ—Ä–∫–∏ (Lookaround Assertions)

–ü—Ä–æ–≤–µ—Ä–∫–∏ (Lookarounds) ‚Äî —ç—Ç–æ –ø—Ä–æ–≤–µ—Ä–∫–∏ –Ω—É–ª–µ–≤–æ–π –¥–ª–∏–Ω—ã. –û–Ω–∏ –ø—Ä–æ–≤–µ—Ä—è—é—Ç –Ω–∞–ª–∏—á–∏–µ (–∏–ª–∏ –æ—Ç—Å—É—Ç—Å—Ç–≤–∏–µ) –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–≥–æ —à–∞–±–ª–æ–Ω–∞ –¥–æ –∏–ª–∏ –ø–æ—Å–ª–µ —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–∏ –∫—É—Ä—Å–æ—Ä–∞.

–ì–ª–∞–≤–Ω–∞—è –æ—Å–æ–±–µ–Ω–Ω–æ—Å—Ç—å: –ö—É—Ä—Å–æ—Ä –Ω–µ —Å–¥–≤–∏–≥–∞–µ—Ç—Å—è –ø–æ—Å–ª–µ –ø—Ä–æ–≤–µ—Ä–∫–∏. –î–≤–∏–∂–æ–∫ "—Å–º–æ—Ç—Ä–∏—Ç" –≤–ø–µ—Ä–µ–¥ –∏–ª–∏ –Ω–∞–∑–∞–¥, —É–±–µ–∂–¥–∞–µ—Ç—Å—è –≤ –Ω–∞–ª–∏—á–∏–∏ —à–∞–±–ª–æ–Ω–∞, –∏ –ø—Ä–æ–¥–æ–ª–∂–∞–µ—Ç –æ—Å–Ω–æ–≤–Ω–æ–π –ø–æ–∏—Å–∫ —Ä–æ–≤–Ω–æ —Å —Ç–æ–π –∂–µ –ø–æ–∑–∏—Ü–∏–∏, –≥–¥–µ –æ—Å—Ç–∞–Ω–æ–≤–∏–ª—Å—è –ø–µ—Ä–µ–¥ –ø—Ä–æ–≤–µ—Ä–∫–æ–π.

–°–∏–Ω—Ç–∞–∫—Å–∏—Å –ù–∞–∑–≤–∞–Ω–∏–µ –ß—Ç–æ –ø—Ä–æ–≤–µ—Ä—è–µ—Ç
(?=...) Positive Lookahead –°–ø—Ä–∞–≤–∞ –µ—Å—Ç—å —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ
(?!...) Negative Lookahead –°–ø—Ä–∞–≤–∞ –Ω–µ—Ç —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è
(?<=...) Positive Lookbehind –°–ª–µ–≤–∞ –µ—Å—Ç—å —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ
(?<!...) Negative Lookbehind –°–ª–µ–≤–∞ –Ω–µ—Ç —Å–æ–≤–ø–∞–¥–µ–Ω–∏—è

4.1. –ü–æ–∑–∏—Ç–∏–≤–Ω–∞—è –æ–ø–µ—Ä–µ–∂–∞—é—â–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (Positive Lookahead) (?=...)

–£—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç, —á—Ç–æ —Å—Ä–∞–∑—É –ø–æ—Å–ª–µ —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–∏ –¥–æ–ª–∂–µ–Ω –Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è —É–∫–∞–∑–∞–Ω–Ω—ã–π —à–∞–±–ª–æ–Ω.

# –ù–∞–π—Ç–∏ "Petr" –∏–ª–∏ "Sergey", —Ç–æ–ª—å–∫–æ –µ—Å–ª–∏ –∑–∞ –Ω–∏–º–∏ –∏–¥–µ—Ç " Ivanov"
names = "Petr Ivanov, Ivan Petrov, Sergey Ivanov"
# \w+ –Ω–∞—Ö–æ–¥–∏—Ç —Å–ª–æ–≤–æ, (?= Ivanov) –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, —á—Ç–æ –¥–∞–ª—å—à–µ –ø—Ä–æ–±–µ–ª –∏ Ivanov
pattern = r"\w+(?= Ivanov)" 
print(re.findall(pattern, names)) # ['Petr', 'Sergey']
# –ó–∞–º–µ—Ç—å—Ç–µ: " Ivanov" –ù–ï –ø–æ–ø–∞–ª –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç

4.2. –ù–µ–≥–∞—Ç–∏–≤–Ω–∞—è –æ–ø–µ—Ä–µ–∂–∞—é—â–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (Negative Lookahead) (?!...)

–£—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç, —á—Ç–æ —Å—Ä–∞–∑—É –ø–æ—Å–ª–µ —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–∏ –Ω–µ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ —à–∞–±–ª–æ–Ω–∞.

# –ù–∞–π—Ç–∏ –∏–º—è, –∑–∞ –∫–æ—Ç–æ—Ä—ã–º –ù–ï —Å–ª–µ–¥—É–µ—Ç " Ivanov"
names = "Petr Ivanov, Ivan Petrov, Sergey Ivanov"
pattern = r"\w+(?! Ivanov)"
print(re.findall(pattern, names)) 
# ['Ivan', 'Petrov', 'Ivanov', 'Ivanov'] 
# (–û–±—Ä–∞—Ç–∏—Ç–µ –≤–Ω–∏–º–∞–Ω–∏–µ: Petrov, Ivanov, Ivanov —Ç–æ–∂–µ –ø–æ–ø–∞–ª–∏, —Ç–∞–∫ –∫–∞–∫ –∑–∞ –Ω–∏–º–∏ –Ω–µ –∏–¥–µ—Ç " Ivanov")

4.3. –ü–æ–∑–∏—Ç–∏–≤–Ω–∞—è —Ä–µ—Ç—Ä–æ—Å–ø–µ–∫—Ç–∏–≤–Ω–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (Positive Lookbehind) (?<=...)

–£—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç, —á—Ç–æ –Ω–µ–ø–æ—Å—Ä–µ–¥—Å—Ç–≤–µ–Ω–Ω–æ –ø–µ—Ä–µ–¥ —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–µ–π –¥–æ–ª–∂–µ–Ω –Ω–∞—Ö–æ–¥–∏—Ç—å—Å—è —É–∫–∞–∑–∞–Ω–Ω—ã–π —à–∞–±–ª–æ–Ω.
–í–∞–∂–Ω–æ: –í Python `re` –¥–ª–∏–Ω–∞ —à–∞–±–ª–æ–Ω–∞ –≤ lookbehind –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–π.

# –ù–∞–π—Ç–∏ —Ü–µ–Ω—É (—á–∏—Å–ª–æ), –ø–µ—Ä–µ–¥ –∫–æ—Ç–æ—Ä–æ–π —Å—Ç–æ–∏—Ç –∑–Ω–∞–∫ $
prices = "$100, ‚Ç¨200, $50"
pattern = r"(?<=\$)\d+"
print(re.findall(pattern, prices)) # ['100', '50']

4.4. –ù–µ–≥–∞—Ç–∏–≤–Ω–∞—è —Ä–µ—Ç—Ä–æ—Å–ø–µ–∫—Ç–∏–≤–Ω–∞—è –ø—Ä–æ–≤–µ—Ä–∫–∞ (Negative Lookbehind) (?<!...)

–£—Ç–≤–µ—Ä–∂–¥–∞–µ—Ç, —á—Ç–æ –Ω–µ–ø–æ—Å—Ä–µ–¥—Å—Ç–≤–µ–Ω–Ω–æ –ø–µ—Ä–µ–¥ —Ç–µ–∫—É—â–µ–π –ø–æ–∑–∏—Ü–∏–µ–π –Ω–µ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å —É–∫–∞–∑–∞–Ω–Ω–æ–≥–æ —à–∞–±–ª–æ–Ω–∞.

# –ù–∞–π—Ç–∏ —Å–ª–æ–≤–æ "bar", –ø–µ—Ä–µ–¥ –∫–æ—Ç–æ—Ä—ã–º –ù–ï–¢ "foo"
text = "foobar bar chocolatebar"
pattern = r"(?<!foo)bar"
print(re.findall(pattern, text)) # ['bar', 'bar'] (–≤—Ç–æ—Ä–æ–π bar –∏–∑ chocolatebar)
